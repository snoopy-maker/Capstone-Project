# Check for missing value - NaN and '' for each column
def check_missing_value(df):
    """
    PURPOSE: Check for NaN and '' for every record
    INPUT:
        df (dataset) - a dataset with user event records
    OUTPUT:
        n/a
    """
    for col in df.columns:
        missing_count = df.filter(df[col].isNull() | (df[col] == "") | isnan(df[col])).count()
        if missing_count > 0:
            print(f"{col} = {missing_count}")

# Extract keyword from userAgent variable and map it to respective browser name
def get_browser_udf(user_agent):
    # Map the browser by keywords
    if "Firefox" in user_agent:
        return "Firefox"
    elif "Safari" in user_agent and "Chrome" in user_agent:
        return "Chrome"
    elif "Safari" in user_agent:
        return "Safari"
    elif "Trident" in user_agent:
        return "IE"
    else:
        return np.NaN

# Register the UDF with the correct return type

# Define a UDF to extract and map the platform
def get_platform_udf (user_agent):
    # Map the browser by keywords
    if "compatible" in user_agent or "Windows" in user_agent:
        return "Windows"
    elif "iPad" in user_agent:
        return "iPad"
    elif "iPhone" in user_agent:
        return "iPhone"
    elif "X11" in user_agent:
        return "Linux"
    elif "Macintosh" in user_agent:
        return "Mac"
    else:
        return np.NaN

def get_col_names(features):
    """
    PURPOSE: Get the names of numeric columns from the given Spark DataFrame features.
    INPUT
        - features: a Spark DataFrame with a schema attribute.
    OUTPUT
        - num_cols: a list of column names representing numeric columns.
    """
    num_cols = []

    for field in features.schema.fields:
        # Assuming that StringType() is used to identify non-numeric columns
        if field.dataType != StringType():
            num_cols.append(field.name)

    if 'label' in num_cols:
        num_cols.remove('label')

    return num_cols

def build_model (classifier, param):
    """
    PURPOSE: Build and configure a machine learning model pipeline with cross-validation
    INPUT
        classifier - the machine learning classifier to be used in the pipeline.
        paramgrid - the parameter grid for hyperparameter tuning during cross-validation.
        features - the features to be used in the model, assumed to be defined elsewhere.
    OUTPUT
        model - the configured CrossValidator model
    """
    # Define a VectorAssembler to combine input features into a single vector column
    assembler = VectorAssembler(inputCols = get_col_names(features), outputCol = 'input_features')
    #assembler = VectorAssembler(inputCols = feature_columns, outputCol = 'input_features')
    
    # Define a Normalizer to scale the input features
    scaler = Normalizer(inputCol="input_features", outputCol="scaled_features")
    
    # Create a Pipeline with the specified stages: VectorAssembler, Normalizer, and the classifier
    pipeline = Pipeline(stages=[assembler, scaler, classifier])
    
    # Create a CrossValidator with the pipeline, parameter grid, evaluator, and number of folds
    model = CrossValidator(
        estimator=pipeline,
        estimatorParamMaps=param,
        evaluator=MulticlassClassificationEvaluator(labelCol='label', metricName='f1'),
        numFolds=3
    )
    
    return model

def get_model_metrics (train_model, classifier_name):
    """
    PURPOSE: Calculate and print basic classification model metrics
    INPUT
        train_model - the trained machine learning model
        classifier_name - the name of classifier
    OUTPUT
        none
    """
    true_positive = (train_model.filter(train_model.label == train_model.prediction).count())
    total_no_predictions  = (train_model.count())
    
    print(f"{classifier_name} Model - Number of Correct Predictions: {true_positive}")
    print(f"{classifier_name} Model - Total Number of Predictions: {total_no_predictions}")
    print(f"{classifier_name} Model - Accuracy: {true_positive / total_no_predictions}")

def predict_data (fit_model, classifier_name):
    """
    PURPOSE: Make predictions using a trained classifier, evaluate performance on the model on test dataset, 
             then compute and print f1 score for specified classifier.
    INPUT
        classifier (string) - the name of classifier
        fit_model - trained machine learning model
    OUTPUT
        none
    """
        
    # Transform the test data using the trained model
    pred = fit_model.transform(test)
    
    # Display model metrics based on predictions
    get_model_metrics(pred, classifier_name)
    
    # Create a MulticlassClassificationEvaluator to evaluate the F1 score
    evaluator = MulticlassClassificationEvaluator(predictionCol="prediction", labelCol="label")
    
    # Evaluate and retrieve the F1 score
    f1_score = evaluator.evaluate(pred, {evaluator.metricName:"f1"})
    print("{} f1: {}".format(classifier_name, f1_score))

Conclusion¶

By comparing accuracy and f1 score of all trained models, clearly the winning model for churn prediction is Logistic Regression model as it achieves the highest accuracy of 0.8235 and f1 score of 0.7797 although it takes slightly longer training time about 1037s.

We did a hyperparameter tuning by performing a grid search with different values for the following parameters with the hope of improving accuracy and F1-score.

  • regParam(0.01, 0.1, 1.0)
  • elasticNetParam (0.0, 0.5, 1.0)
  • maxIter (10, 20, 30)

Unfortunately, the accurancy and F1-score reduced to 0.7647 and 0.7062 after lowering maxIter from 100 to 30 as the model may not have enough iterations to converge to the optimal set of coefficients. In addition, the running cost of this tuned model increased significantly with triple training time compared to initial model. Hence, we conclude that we should stick to its default parametes while building Logistic Regression model for this small dataset.
